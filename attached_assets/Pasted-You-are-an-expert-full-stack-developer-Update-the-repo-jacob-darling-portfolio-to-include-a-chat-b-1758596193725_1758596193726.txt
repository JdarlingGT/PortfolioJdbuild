2. API Route â€” src/app/api/chat/route.ts
import { NextResponse } from "next/server";
import OpenAI from "openai";
import portfolioData from "@/data/portfolioData.json";

const client = new OpenAI({
  apiKey: process.env.OPENAI_API_KEY,
});

export async function POST(req: Request) {
  try {
    const { messages } = await req.json();

    const context = `
You are Jacob Darlingâ€™s marketing portfolio AI agent.
Answer questions concisely and professionally using the following data:

### Employment
${portfolioData.employment.map(e =>
  `- ${e.role} at ${e.company} (${e.years}): ${e.highlights.join(" ")}`
).join("\n")}

### Bearcave Marketing Projects
${portfolioData.bearcaveProjects.map(p =>
  `- ${p.client} (${p.website}): ${p.services.join(", ")}. Outcome: ${p.outcome}`
).join("\n")}

### Capabilities
${portfolioData.capabilities.join(", ")}
    `;

    const completion = await client.chat.completions.create({
      model: "gpt-4o-mini",
      messages: [
        { role: "system", content: context },
        ...messages,
      ],
    });

    return NextResponse.json({
      reply: completion.choices[0].message?.content || "",
    });
  } catch (err) {
    console.error(err);
    return NextResponse.json({ error: "Something went wrong." }, { status: 500 });
  }
}

3. Chat Component â€” src/components/ChatAgent.tsx
"use client";

import { useState } from "react";

export default function ChatAgent() {
  const [messages, setMessages] = useState<{ role: string; content: string }[]>([]);
  const [input, setInput] = useState("");
  const [loading, setLoading] = useState(false);

  async function sendMessage() {
    if (!input.trim()) return;

    const newMessages = [...messages, { role: "user", content: input }];
    setMessages(newMessages);
    setInput("");
    setLoading(true);

    try {
      const res = await fetch("/api/chat", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({ messages: newMessages }),
      });

      const data = await res.json();
      setMessages([...newMessages, { role: "assistant", content: data.reply }]);
    } catch (e) {
      setMessages([...newMessages, { role: "assistant", content: "Error: failed to connect." }]);
    } finally {
      setLoading(false);
    }
  }

  return (
    <div className="w-full max-w-lg mx-auto border rounded-2xl shadow-lg p-4 bg-white dark:bg-gray-900">
      <h2 className="text-xl font-bold mb-2 text-gray-800 dark:text-gray-100">
        Chat with My AI Agent
      </h2>
      <div className="h-64 overflow-y-auto border rounded-lg p-2 bg-gray-50 dark:bg-gray-800">
        {messages.map((m, i) => (
          <div key={i} className={`mb-2 ${m.role === "user" ? "text-right" : "text-left"}`}>
            <span
              className={`inline-block px-3 py-2 rounded-lg ${
                m.role === "user"
                  ? "bg-blue-500 text-white"
                  : "bg-gray-300 dark:bg-gray-700 text-gray-900 dark:text-gray-100"
              }`}
            >
              {m.content}
            </span>
          </div>
        ))}
      </div>

      <div className="flex mt-2 gap-2">
        <input
          type="text"
          value={input}
          onChange={(e) => setInput(e.target.value)}
          placeholder="Ask me about Jacobâ€™s work..."
          className="flex-1 border rounded-lg px-3 py-2 dark:bg-gray-800 dark:text-white"
        />
        <button
          onClick={sendMessage}
          disabled={loading}
          className="bg-blue-600 text-white px-4 py-2 rounded-lg hover:bg-blue-700 disabled:opacity-50"
        >
          {loading ? "..." : "Send"}
        </button>
      </div>
    </div>
  );
}
